function withMouse(WrappedComponent) {
  class Mouse extends React.Component {
    state= {x:0,y:0}

      this.setState({x:e.clientX,y:e.clientY})
    }

      window.removeEventListener("mousemove",this.handle);

    }

    // 设置displayName
  Mouse.displayName =`withMouse${getDisplayName(WrappedComponent)}`
  return Mouse
}

减轻state #

只存储跟组件渲染相关的数据（比如：count / 列表数据 / loading 等） 不用做渲染的数据不要放在state中，比如定时器id等 对于这种需要在多个方法中用到的数据，应该放在this中

子组件没有任何变化时也会重新渲染 如何避免不必要的重新渲染 解决方案: 使用钩子函数shouldComponentUpdate(nextProps, nextState) 通过返回值决定该组件是否重新渲染，返回true表示重新渲染， false表示不重新渲染 触发时机:更新阶段的钩子函数，组件重新渲染前执行 shouldComponentUpdate -> render

js
export default class App extends Component {
state = {
    count:0
  }
handle= ()=>{
    this.setState(state=>{
      return {
        count:state.count+1
      }
    })
  }
// 钩子函数 
shouldComponentUpdate(nextProps, nextState){
  console.log('最新state',nextState);
  // console.log('最新Props',nextProps);
  console.log('this.state',this.state);
  return true
  // 返回false 阻止组件重新渲染
}

render() {
    console.log('组件更新了');
    return (
      <div>
      <h1>计数器:{this.state.count}</h1>
      <button onClick={this.handle}>+1</button>
      </div>
    )
  }
}

useEffect(处理副作用)和useLayoutEffect(同步执行副作用)

js
import React, { useEffect, useState } from 'react'
export default function App() {
  const [name,setName] = useState('kerwss');
  useEffect(() => {
   setName(name.substring(0,1).toUpperCase()+name.substring(1))
  }, [name]) // 传空数组 不依赖于任何东西
  return (
    <div>
      app-{name}
      <button onClick={()=>{
        setName('xiaomign1')
      }}>点我</button>
    </div>
  )
}

实际使用时，如果想要避免页面抖动，可以把需要操作DOM的代码放在useLayoutEffect里，在这里做点DOM操作， 这些dom修改会和 react 做出的更改一起被一次性渲染到屏幕上，只有一次回流、重绘的代价。

useCallback(记忆函数) 防止因为组件被重新渲染，导致方法被重新创建，起到缓存作用，第二个参数变化了，才重新声明一次

js
var handclick=useCallback(()=>{
  console.log(name);
},[name])
<button onClick={()=>handclick()}>hi</button>
// nam改变后，这个函数才会重新声明一次  
// 如果传入空数组，那么就是第一次创建后就被缓存，如果name后期改变了，拿到了还是老的name
// 如果不传入第二个参数，每次都会重新声明一次，拿到的就是最新的name

useMemo 记忆组件
useMemo是用来缓存计算属性的。 直接在渲染的时候就执行，在DOM区域被当作属性值一样去使用。 会执行第一个函数并且将函数执行结果返回给你 经过函数计算得到一个确定的值，比如记忆组件 useMemo(p1,p2) 第一个参数填写需要缓存的计算属性 第二个写 依赖 useMemo每次在需要重新计算时去比较是否被更改，只有当依赖改变了被useMemo保护的函数才会重新执行，否则拒绝重新执行，直接返回旧的计算属性值

被保护后只有在组件初始化、依赖改变时才会执行

  • useMemo 是不是用的越多越好

    • 不是 因为缓存不是免费的,所有被useMemo保护的函数都会被加入useMemo的工作队列
    • 在组件进行渲染并且组件内使用了useMemo之后，为了校验该组件内被保护的这个计算属性是否需要重新计算， 它会先去useMemo的工作队列中找到该函数，然后去校验该函数的依赖 是否被更改
    • 存在的核心问题 寻找到需要校验的计算属性和进行校验这2step都需要成本
    • 结论: 当大量的使用useMemo后，可能会造成反向优化

  • 在什么情况下使用

    1. 某一个计算属性需要大量的计算时

Flux & Redux #

用一个单独的常量状态树(state对象)保存这一整个应用的状态，这个对象不能直接被改变， 当一些数据发生变化，一个新的对象就会被创建(使用actions reducer),这样就可以进行数据追踪 Redux 主要作用: 应用状态的管理
state以单一对象存储在store对象中
state只读(每次都返回一个新的对象) 使用纯函数reducer执行state更新

redux 工作流 #

工作流流程 通过UI界面或者状态 点击 例如button onclick事件去 派发一个dispatch
dispatch 一个(例如 dispatch({type: 'counter/incremented'}) )action 给store 然后store通过以前的状态和 当前的动作来操作reducer 并将返回值保存为新的state store通知订阅的UI store已更新

每个需要存储数据的 UI 组件都会检查它们需要的状态部分是否已更改。 每个看到其数据已更改的组件都会强制使用新数据重新渲染，因此它可以更新屏幕上显示的内容

js
// 1.创建redux
// 2.createStore(reducer)
// import { legacy_createStore as createStore} from 'redux'

const reducer = (prevState={show:true},action)=>{
  let newState = {...prevState};
  console.log('上次的state值',newState);
  switch(action.type){
    case 'xx-hidden':
      newState.show = false
      return newState;
    case 'xx-show':
      newState.show = true
      return newState
    default:
      return prevState
  }
}
const  store = createTianStore(reducer)

// redux原理  
/**
 * store.dispatch
 * store.subscribe
 * store.getState
 */
function createTianStore(reducer){
  var list = [];
  var state = reducer(undefined,{})

  function dispatch(action){
    state = reducer(state,action)
    for(var i in list){
      list[i] && list[i]()
    }
  }
  function subscribe(callback){
    list.push(callback)
  }
  function getState(){
    return state
  }
  return {
    dispatch,
    subscribe,
    getState
  }
}
export default store

React.lazy #

允许延迟加载组件，直到它们真正需要渲染为止。 这可以帮助您优化应用程序的性能，特别是在初始加载时。

使用步骤

  1. 使用React.lazy函数加载组件
js
const MyComponent = React.lazy(()=> import('./MyComponent'))
  1. 使用React.Suspense组件包装lazy组件
js
<Suspense fallback={<div>Loading...</div>}>
  <MyComponent />
</Suspense>
  1. 在组件中渲染lazy组件 React.Suspense 组件允许您指定在组件加载完成之前显示的内容。fallback 属性可用于指定要显示的内容。

WARNING

React.lazy / React.Suspense 需要版本 > 16.6

js
import React, { lazy, Suspense } from 'react';

const OtherComponent = lazy(() => import('./OtherComponent'));

function MyComponent() {
  return (
    <Suspense fallback={<div>Loading...</div>}>
      <OtherComponent />
    </Suspense>
  );
}

export default MyComponent;

在MyComponent组件被渲染时，OtherComponent不会立即被加载，而是在其需要渲染时才加载，可以减少首次加载的资源消耗，提升性能。

forwardRef #

接受一个函数作为参数，该函数接受props和ref两个参数，并返回一个react元素 调用该函数时，将ref附加到返回的元素上

使用 forwardRef 可以让函数组件访问 DOM 节点或子组件的 refs，这使得函数组件能够与类组件一样完成高级操作。

js
import React, { useRef, forwardRef } from 'react';

const Child = forwardRef((props, ref) => {
    return (
        <div style={{ background: "red" }}>
            <input ref={ref} defaultValue="22222" />
        </div>
    );
});

function App() {
    const mytext = useRef(null);

    const handleClick = () => {
        console.log(mytext.current);
        mytext.current.value = "";
        mytext.current.focus();
    };

    return (
        <div>
            <button onClick={handleClick}>获取焦点</button>
            <Child ref={mytext} />
        </div>
    );
}

export default App;
